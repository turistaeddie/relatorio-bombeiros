    <h2>Relatório Gerado:</h2>
    <textarea id="relatorio" readonly></textarea>

    <script>
        function gerarRelatorio() {
            const tipo = document.getElementById("tipo").value;
            const endereco = document.getElementById("endereco").value;
            const ids = document.getElementById("ids").value;
            const descricao = document.getElementById("descricao").value;

            let relatorio = `🚒 **Relatório de Ocorrência** 🚒\n\n`;
            relatorio += `🔹 **Tipo de Ocorrência:** ${tipo}\n`;
            relatorio += `📍 **Endereço:** ${endereco}\n`;

            if (ids.trim() !== "") {
                relatorio += `🆔 **ID(s) da(s) Vítima(s):** ${ids}\n`;
            }

            if (descricao.trim() !== "") {
                relatorio += `📝 **Descrição:** ${descricao}\n`;
            }

            relatorio += `✅ **Atendimento concluído com sucesso.**`;

            document.getElementById("relatorio").value = relatorio;
        }
    </script>
